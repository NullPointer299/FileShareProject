CREATE TABLE USERS(ID VARCHAR(20) PRIMARY KEY, FIRST_NAME VARCHAR(20), LAST_NAME VARCHAR(20), PASSWORD TEXT, PUBLIC INT, INDEX(ID))ENGINE=InnoDB;
CREATE TABLE FILES(ID VARCHAR(20) REFERENCES USERS(ID), PARENT_PATH TEXT, NAME TEXT, PUBLIC INT, PRIMARY KEY(ID, PARENT_PATH(128), NAME(128)), INDEX(ID))ENGINE=InnoDB;
CREATE TABLE FAVORITE(ID VARCHAR(20) REFERENCES USERS(ID) ON DELETE CASCADE, FAV_ID VARCHAR(20), PRIMARY KEY(ID, FAV_ID), INDEX(ID))ENGINE=InnoDB;
